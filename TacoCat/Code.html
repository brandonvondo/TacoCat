Behind the Taco

Here is the code behind this application

$("#btntaco").on("click", function () { /*creating an unnamed function with a click event on our start button*/
if ($("#inptaco").val() != "") { /*super simple check to see if the input field is blank*/
let orig = $("#inptaco").val(); /*holding our input for a cleaner output after we get our results*/
let origverse = orig.split("").reverse().join("") /*reverse the input with all of punctuation and capitalization the same*/
let word = $("#inptaco").val().toLowerCase() /*taking the value of our input field and forcing it to lowercase*/
.replace(/\s/g, '')/*replacing spaces*/.replace(/\./g, '');/*replacing .*/

//next line is creating a variable to hold the new results with leaving word the same
let revWord = word.split("")/*acting on variable word as if it were an array + splitting letters*/.reverse().join("");

if (word === revWord) { /*checking if our changed variable is equal to our original word*/
$("#tacout").html(`Your original input was "${orig}" reversed it is "${origverse}" and after being cleaned it up is ${revWord} so it definitely is a Palindrome.`)
}
else { /*if it isn't we are out putting a failure message*/
$("#tacout").html(`"${orig}" reversed is ${origverse} is it a Palindrome? You tell me. (No)`)
}
}
else $("#tacout").html(`Please enter a valid word or phrase.`)
});

$("#btnclear").on("click", function () { /*simple button to clear the results and the input field*/
$("#inptaco").val("");
$("#tacout").text("");
});
